```typescript
import { BrevoClient } from "@getbrevo/brevo";
import { buildEmailHtml } from "./buildEmailHtml";

const brevo = new BrevoClient({
  apiKey: process.env.BREVO_API_KEY!,
});

export interface SendLetterEmailParams {
  recipientEmail: string;
  politicianName: string;
  politicianTitle: string | null;
  politicianPostalAddress: string;
  letterText: string;
  issueText: string;
}

export async function sendLetterEmail(
  params: SendLetterEmailParams
): Promise<{ success: boolean; messageId?: string }> {
  try {
    const result = await brevo.transactionalEmails.sendTransacEmail({
      subject: "Dein Brief nach Berlin ist fertig",
      htmlContent: buildEmailHtml(params),
      sender: {
        name: "Brief nach Berlin",
        email: process.env.BREVO_SENDER_EMAIL || "brief@brief-nach-berlin.de",
      },
      to: [{ email: params.recipientEmail }],
    });
    return { success: true, messageId: result.messageId };
  } catch (error) {
    console.error("[brief-nach-berlin] Brevo send failed:", error);
    return { success: false };
  }
}
```

Key decisions per D-01/D-02/D-07:
- Brevo API (not SMTP) per D-02
- Subject line "Dein Brief nach Berlin ist fertig" per D-07
- Sender name "Brief nach Berlin" with configurable email via BREVO_SENDER_EMAIL env var (fallback to brief@brief-nach-berlin.de)
- Graceful error handling: logs and returns `{ success: false }` on failure (does not throw)

|----------|-------------|
| Server action -> Brevo API | API key crosses from server env to external service |
| User input -> email HTML | issueText and letterText are embedded in HTML template |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Information Disclosure | sendLetterEmail.ts | mitigate | BREVO_API_KEY stored as env var only, never prefixed with NEXT_PUBLIC_, server-side file only (not imported by client components) |
| T-03-02 | Tampering | buildEmailHtml.ts | mitigate | letterText and issueText are embedded in pre-wrap block with HTML entity escaping. Convert `<`, `>`, `&` to entities before embedding in HTML to prevent HTML injection in email. |
| T-03-03 | Tampering | buildEmailHtml.ts (URL param) | mitigate | issueText is URL-encoded via encodeURIComponent before embedding in ?text= link. Truncated at 1500 chars. |
| T-03-04 | Denial of Service | sendLetterEmail.ts | accept | Brevo free tier 300/day limit acts as natural rate limit. Sufficient for v1 validation. No additional rate limiting needed at this layer. |
