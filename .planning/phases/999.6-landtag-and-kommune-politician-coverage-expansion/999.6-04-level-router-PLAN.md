    import { z } from "zod";
    import { mistral } from "@/lib/mistral";
    import taxonomy from "../../../data/zustaendigkeit-taxonomie.json";

    export const RoutingResultSchema = z.object({
      level: z.enum(["Bund", "Land", "Gemeinde"]),
      confidence: z.enum(["high", "medium", "low"]),
      reasoning: z.string().min(1).max(500),
    });
    export type RoutingResult = z.infer<typeof RoutingResultSchema>;

    export class LevelRouterError extends Error {
      constructor(message: string, public cause?: unknown) {
        super(message);
        this.name = "LevelRouterError";
      }
    }

    function buildSystemPrompt(): string {
      return [
        "Du bist Experte für deutschen Föderalismus.",
        "Ordne das Anliegen einer politischen Ebene zu: Bund, Land oder Gemeinde.",
        "Wähle immer die konkret handlungsfähige Ebene, nicht nur die rechtliche Rahmenebene.",
        "",
        "BUND - ausschließlich:",
        ...taxonomy.bund.exclusive.map((t: string) => `- ${t}`),
        "",
        "BUND - überwiegend (konkurrierend):",
        ...taxonomy.bund.concurrent_dominant.map((t: string) => `- ${t}`),
        "",
        "LAND - ausschließlich:",
        ...taxonomy.land.exclusive.map((t: string) => `- ${t}`),
        "",
        "GEMEINDE - ausschließlich:",
        ...taxonomy.gemeinde.exclusive.map((t: string) => `- ${t}`),
        "",
        "Antworte ausschließlich als JSON: {\"level\":\"Bund|Land|Gemeinde\",\"confidence\":\"high|medium|low\",\"reasoning\":\"kurze Begründung auf Deutsch, max 200 Zeichen\"}",
      ].join("\n");
    }

    export async function routeToLevel(anliegen: string): Promise<RoutingResult> {
      if (!anliegen || anliegen.trim().length < 3) {
        throw new LevelRouterError("Anliegen zu kurz für Routing");
      }
      let raw: string;
      try {
        const res = await mistral.chat.complete({
          model: "mistral-small-latest",
          temperature: 0.1,
          responseFormat: { type: "json_object" },
          messages: [
            { role: "system", content: buildSystemPrompt() },
            { role: "user", content: `Anliegen: ${anliegen}` },
          ],
        });
        const content = res.choices?.[0]?.message?.content;
        raw = typeof content === "string" ? content : JSON.stringify(content);
      } catch (err) {
        throw new LevelRouterError("Mistral routing call failed", err);
      }

      // Strip ```json fences if present (mirrors generateLetter.ts pattern)
      const cleaned = raw.replace(/```json\s*/g, "").replace(/```\s*$/g, "").trim();

      let parsed: unknown;
      try {
        parsed = JSON.parse(cleaned);
      } catch (err) {
        throw new LevelRouterError(`Mistral returned non-JSON: ${cleaned.slice(0, 200)}`, err);
      }

      const result = RoutingResultSchema.safeParse(parsed);
      if (!result.success) {
        throw new LevelRouterError(`Mistral output failed schema: ${result.error.message}`);
      }
      return result.data;
    }
    ```

         const enrichment = (plzBundeslandJson as Record<string, {

         // Bund: existing path
         const bundWahlkreisIds = (plzMapping[plz] ?? []);
         const bund = politiciansCache.bundestag.filter(p => bundWahlkreisIds.includes(p.wahlkreisId));

         // Land: only if covered Land
         let land: Politician[] = [];
         const landSupported = bundeslandKey != null && COVERED_LAENDER.has(bundeslandKey);
         if (landSupported && bundeslandKey) {
           const stimmkreisIds = STIMMKREIS_BY_LAND[bundeslandKey][plz] ?? [];
           land = politiciansCache.landtag.filter(
             p => p.bundeslandKey === bundeslandKey && stimmkreisIds.includes(p.wahlkreisId)

         }

         return {
           bundeslandKey,
           bundeslandName,
           byLevel: { Bund: bund, Land: land, Gemeinde: [] },
           coverage: { landSupported },
         };
       }
