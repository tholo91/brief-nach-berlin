    ```tsx
    import type { PoliticalLevel } from "@/lib/types/politician";

    interface LevelBadgeProps {
      level: PoliticalLevel;
      className?: string;
    }

    const LEVEL_COPY: Record<PoliticalLevel, { label: string; description: string; classes: string }> = {
      Bund: {
        label: "Bund",
        description: "Bundestag, Berlin",
        classes: "bg-waldgruen/15 text-waldgruen-dark border-waldgruen/30",
      },
      Land: {
        label: "Land",
        description: "Landtag",
        classes: "bg-airmail-rot/10 text-airmail-rot border-airmail-rot/30",
      },
      Kommune: {
        label: "Gemeinde",
        description: "Rathaus, Stadtrat",
        classes: "bg-warmgrau/15 text-warmgrau border-warmgrau/40",
      },
    };

    export function LevelBadge({ level, className }: LevelBadgeProps) {
      const meta = LEVEL_COPY[level];
      return (
        <span
          title={meta.description}
          className={[
            "inline-flex items-center gap-1 font-body text-[11px] font-semibold uppercase tracking-wide px-2 py-0.5 rounded border",
            meta.classes,
            className ?? "",
          ].join(" ")}
        >
          <span aria-hidden="true">●</span>
          Ebene: {meta.label}
        </span>
      );
    }
    ```

    - The mapping `Kommune → "Gemeinde"` is intentional: the type uses "Kommune" but the user-facing label per D7 is "Gemeinde" (matches the taxonomy file).
    - No em dashes (`grep -c "—"` must return 0).
    - No `"use client"` needed — this is a pure render component.
