    ```tsx
    import type { PoliticalLevel } from "@/lib/types/politician";
    import { Landmark, MapPin, Home } from "lucide-react";

    interface LevelOverrideChipsProps {
      primaryLevel: PoliticalLevel;
      availableLevels: PoliticalLevel[];
      selectedLevel: PoliticalLevel | null; // null = no override, show primary group
      onSelect: (level: PoliticalLevel | null) => void;
      className?: string;
    }

    const LEVEL_META: Record<PoliticalLevel, { label: string; Icon: typeof Landmark }> = {
      Bund: { label: "Bund", Icon: Landmark },
      Land: { label: "Land", Icon: MapPin },
      Kommune: { label: "Gemeinde", Icon: Home },
    };

    export function LevelOverrideChips({
      primaryLevel,
      availableLevels,
      selectedLevel,
      onSelect,
      className,
    }: LevelOverrideChipsProps) {
      const others = availableLevels.filter((l) => l !== primaryLevel);
      if (others.length === 0) return null;
      const isOverriding = selectedLevel !== null && selectedLevel !== primaryLevel;
      return (
        <div className={["flex flex-wrap items-center gap-2 text-sm font-body", className ?? ""].join(" ")}>
          <span className="text-warmgrau">Andere Ebene?</span>
          {others.map((level) => {
            const { label, Icon } = LEVEL_META[level];
            const active = selectedLevel === level;
            return (
              <button
                key={level}
                type="button"
                onClick={() => onSelect(active ? null : level)}
                aria-pressed={active}
                className={[
                  "inline-flex items-center gap-1 px-3 py-1 rounded-full border transition-colors",
                  active
                    ? "bg-waldgruen text-creme border-waldgruen"
                    : "bg-transparent text-waldgruen-dark border-waldgruen/40 hover:bg-waldgruen/10",
                ].join(" ")}
              >
                <Icon size={14} aria-hidden="true" />
                {label}
              </button>
            );
          })}
          {isOverriding && (
            <button
              type="button"
              onClick={() => onSelect(null)}
              className="text-warmgrau underline underline-offset-2 hover:text-waldgruen-dark"
            >
              Zurück zur Empfehlung
            </button>
          )}
        </div>
      );
    }
    ```

    ```tsx
    }
    ```
