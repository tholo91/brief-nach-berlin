```typescript
import type { Politician } from "../types/politician";

export interface RathausRecipient extends Politician {
  isSynthetic: true;           // sentinel — distinguishes from real Abgeordnete
  recipientKind: "stadtverwaltung" | "bezirksamt"; // BE uses bezirksamt
}

export function buildRathausRecipient(args: {

  plz: string;
  bundeslandKey: string;
  bundeslandName: string;
}): RathausRecipient;
```

The synthetic recipient is shaped like a normal `Politician` so it flows through `generateLetter()` unchanged, but with these field values:

| Politician field | Stadtverwaltung value | Notes |
|---|---|---|
| `id` | `-1` (or a sentinel like `-1000 - hash(plz)`) | Distinct from real AW politician IDs |
| `firstName` | `""` (empty) | No personal name |
| `lastName` | `"Stadtverwaltung"` (or `"Bezirksamt"`) | Acts as the recipient label |
| `title` | `null` | No academic title |
| `gender` | `"d"` (neutral) | For Anrede branch in PLAN 06 |
| `formal_title` | `"Stadtverwaltung [Ortsname]"` | Used in letter Anrede + email |
| `party` | `null` | No party |
| `wahlkreisId` | `0` | No constituency |
| `wahlkreisName` | `ortsname` | "Köln, Stadt" |
| `level` | `"Kommune"` | Canonical (review B3) |
| `bundeslandKey` | `bundeslandKey` | From OpenPLZ |
| `postalAddress` | `"Stadtverwaltung [Ortsname]\n[PLZ] [Ortsname]"` | Generic post-tauglich |
| `abgeordnetenwatchUrl` | `null` | Not an AW politician |
| `isDirect` | `false` | n/a for Stadtverwaltung |
| `isSynthetic` | `true` | NEW flag |
| `recipientKind` | `"stadtverwaltung"` or `"bezirksamt"` | NEW flag |

**Berlin special case:**
- For `bundeslandKey === "BE"`: `recipientKind = "bezirksamt"`, `lastName = "Bezirksamt"`, `formal_title = "Bezirksamt [Ortsname]"`. OpenPLZ ortsname for Berlin contains the Bezirk (e.g. "Berlin, Friedrichshain-Kreuzberg"), so this works.

**Hamburg / Bremen re-route:**
- For `bundeslandKey ∈ {"HH", "HB"}`: do NOT build a Kommune recipient. The Routing layer (`submitWizard.ts` per PLAN 04 + 06) re-routes Kommune → Land per D12. PLAN 07 just gracefully returns `null` for these.
